#Konovalets wins in civil war
news_event = {
	id = news.4
	title = news.4.t
	desc = news.4.d
	picture = GFX_news_event_007
	
	major = yes
	
	trigger = {
	    has_global_flag = ukrainian_civil_war
		NOT = { has_global_flag = ucw_over }
		UMR = { has_civil_war = no }
		any_other_country = {
			original_tag = UMR
			has_government = nationalism
		}
		all_country = {
			OR = {
				NOT = { original_tag = UMR }
				has_government = nationalism
			}
		}
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		set_global_flag = konovalets_victory
		set_global_flag = ucw_over
	}
	
	option = {
		name = news.4.a
		trigger = {
		}
	}
}

#Authoritarianism wins in Austria
news_event = {
	id = news.6
	title = news.6.t
	desc = news.6.d
	picture = GFX_news_event_009
	
	major = yes
	
	trigger = {
	    has_global_flag = austrian_civil_war
		NOT = { has_global_flag = austrian_civil_war_over }
		AVS = { has_civil_war = no }
		any_other_country = {
			original_tag = AVS
			has_government = authoritarianism
		}
		all_country = {
			OR = {
				NOT = { original_tag = AVS }
				has_government = authoritarianism
			}
		}
	}
	
	mean_time_to_happen = {
	    days = 2
	}
	
	immediate = {
	    set_global_flag = austrian_civil_war_over
	}
	
	option = {
		name = news.6.a
		trigger = {
		    NOT = { original_tag = AVS }
		    OR = {
			    has_government = socialism
				has_government = trotskism
				has_government = liberalism
				has_government = anarchism
			}
		}
	}
	option = {
		name = news.6.b
		trigger = {
		    NOT = { original_tag = AVS }
		    OR = {
			    has_government = conservatism
				has_government = nationalism
				has_government = authoritarianism
			}
		}
	}
	option = {
		name = news.6.c
		trigger = {
		    original_tag = AVS
		}
	}
}
